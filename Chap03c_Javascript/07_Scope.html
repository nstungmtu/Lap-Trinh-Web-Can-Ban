Scope (phạm vi) trong Javascript

Scope quy định khả năng truy cập của biến trong mã nguồn Javascript, có 3 loại scope:

  • Block scope: các dòng lệnh bên trong cặp dấu {}
  • Function scope: các dòng lệnh bên trong khai báo hàm
  • Global scope: phạm vi toàn bộ trang web.

Block scope

//let
let x = 2;
{
    let x = 3;
    console.log(x);//3
}
console.log(x);//2

//const
const x = 2;
{
    const x = 3;
    console.log(x);//3
}
console.log(x);//2

//var (không hỗ trợ block scope)
var x = 2;
{
    var x = 3;
    console.log(x);//3
}
console.log(x);//3

Function scope


let x = 1;
let y = 1;
function fx(){
    console.log("fx:Giá trị x bên ngoài: " + x);//1
}
function fy(){
    let y = 3;
    console.log("fy:Giá trị y bên trong: " + y);//3
}
function fy2(y){
    console.log("fy2:Giá trị y tham số hàm: " + ++y);
}
function fy3(){
    console.log("fy3:Giá trị y bên ngoài: " + y);
    y=5;
    console.log("fy3:Giá trị y bên ngoài: " + y);
}
fx();//1
fy();//3
fy2(y);//2
fy3();//lúc đầu là 1, lúc sau là 5
console.log("Giá trị y bên ngoài: " + y);//5

Global scope

//nội dung tập tin script1.js
let a = 1;

//nội dung tập tin script2.js
let b = 2;

//nội dung tập tin script3.js
