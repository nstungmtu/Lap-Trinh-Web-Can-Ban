Kiểu dữ liệu trong Javascript

Các kiểu dữ liệu trong Javascript

  1. String - kiểu chuỗi
  2. Number - kiểu số
  3. Bigint - số nguyên lớn
  4. Boolean- true, false
  5. Undefined - không có giá trị
  6. Null - rỗng
  7. Object (object, Array, Date)

//String template
let firstName = "Dolores", lastName = "Abernathy";
console.log(`Hello ${firstName} ${lastName}`);//Hello Dolores Abernathy

let price = 100, VAT = 0.05;
console.log(`Total: ${price * (1 + VAT)}`); //105

let html = `<ul>`;
for(let i = 1; i < 3; i++){
    html += `<li>${i}</li>`;
}
html += `</ul>`;
console.log(html);

//Kiểu Object
const student = {
    firstName:"Dolores", 
    lastName:"Abernathy", 
    class:{
        name:"Software Engineering", 
        year:2023
    }};
for(let i in student){
    console.log(i); //firstName, lastName, class
    console.log(student[i]); //Dolores, Abernathy, ...
}

//Kiểu Array
const cars = ["Toyota", "Chevrolet", "Hyundai"];
for(let i in cars){
    console.log(i);//0,1,2
}
for(let i in cars){
    console.log(cars[i]);//Toyota, Chevrolet, Hyundai
}
for(let i of cars){
    console.log(i);//Toyota, Chevrolet, Hyundai
}
cars.forEach(function(value, index, array){
    console.log(value);//Toyota, Chevrolet, Hyundai
    console.log(index);//0,1,2
    console.log(array);//mảng cars
});
